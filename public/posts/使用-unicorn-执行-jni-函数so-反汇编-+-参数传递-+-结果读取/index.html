Java_com_cyrus_example_unicorn_UnicornActivity_add(JNIEnv *env, jobject thiz, jint a, jint b) {
    return a + b;
}

加载 so 并反汇编

通过 IDA 打开 so 文件可以看到 add 方法的开始地址为 0x0608，结束地址为 0x063C

加载 so 文件并使用 capstone 反汇编 so 中机器码

import capstone

# 加载 libunicorn.so
so_file = "libunicorn.so"

# 指定 add 方法的开始和结束地址
start_address = 0x0608
end_address = 0x063C

# 读取 .so 文件内容
with open(so_file, "rb") as f:
    binary_data = f.read()

# 提取需要反汇编的字节码（包括结束地址）
add_function_data = binary_data[start_address:end_address]

# 初始化 Capstone 反汇编器 (针对 ARM64 架构)
md = capstone.Cs(capstone.CS_ARCH_ARM64, capstone.CS_MODE_ARM)

# 反汇编字节码
for instruction in md.disasm(add_function_data, start_address):
    print(f"0x{instruction.address:x}:\t{instruction.mnemonic}\t{instruction.op_str}")

汇编代码分析

add 方法的汇编代码分析如下：

SUB             SP, SP, #0x20 ; 分配栈空间 0x20 (32 字节)
STR             X0, [SP,#0x20+var_8] ; 保存 X0（env 指针）
STR             X1, [SP,#0x20+var_10] ; 保存 X1（thiz 对象）
STR             W2, [SP,#0x20+var_14] ; 保存 W2（参数 a）
STR             W3, [SP,#0x20+var_18] ; 保存 W3（参数 b）
LDR             W9, [SP,#0x20+var_14] ; 读取参数 a 到 W9
LDR             W10, [SP,#0x20+var_18] ; 读取参数 b 到 W10
MOV             W8, WZR ; W8 = 0（WZR 是 zero 寄存器，表示 0）
SUBS            W9, W8, W9 ; W9 = 0 - a（W9 变成 -a）
SUBS            W10, W8, W10 ; W10 = 0 - b（W10 变成 -b）
ADD             W9, W9, W10 ; W9 = (-a) + (-b) = -(a + b)
SUBS            W0, W8, W9 ; W0 = 0 - W9 = a + b
ADD             SP, SP, #0x20  ; 释放栈空间
RET                     ; 返回

使用 Unicorn 调用 so 中方法

2.1 初始化 Unicorn

初始化 Unicorn 引擎的实例

import unicorn
from unicorn import Uc, UC_ARCH_ARM64, UC_MODE_ARM
from unicorn.arm64_const import *


# 创建 Unicorn 引擎实例 (ARM64)
mu = Uc(UC_ARCH_ARM64, UC_MODE_ARM)

2.2 加载 so 并反汇编 jni 方法

import capstone

# 加载 libunicorn.so
so_file = "libunicorn.so"

# 指定 add 方法的开始和结束地址
start_address = 0x0608

2.3 内存分配与指令写入

# **分配代码区（TEXT 段）**
CODE_BASE = 0x100000  # 假设代码段起始地址
CODE_SIZE = 0x1000  # 4KB
mu.mem_map(CODE_BASE, CODE_SIZE)

# **分配栈区（STACK 段）**
STACK_BASE = 0x400000  # 假设栈基地址
STACK_SIZE = 0x10000  # 64KB
mu.mem_map(STACK_BASE, STACK_SIZE)

# **写入指令**
mu.mem_write(CODE_BASE, add_function_data)

2.4 参数传递、初始化 SP 和 PC

设置 x2 和 x3 寄存器值为 1，计算 1 + 1 的结果。

# **设置寄存器初始值**
mu.reg_write(UC_ARM64_REG_X0, 0)  # env (ignored)
mu.reg_write(UC_ARM64_REG_X1, 0)  # thiz (ignored)
mu.reg_write(UC_ARM64_REG_X2, 1)  # a = 1
mu.reg_write(UC_ARM64_REG_X3, 1)  # b = 1

# 栈指针（对齐）
mu.reg_write(UC_ARM64_REG_SP, STACK_BASE + STACK_SIZE - 4) # 使 SP 从栈的顶部往下移动 4 字节，以 预留一点空间，避免越界错误。

# **设置 PC**
mu.reg_write(UC_ARM64_REG_PC, CODE_BASE)

注意：如果参数个数超过一定数量时，ARM 和 ARM64 参数传递时的处理如下

ARM (AArch32) 参数传递规则

  1. 前 4 个参数（整数/指针）使用 R0-R3 传递。

  2. 第 5 个及后续参数 存放在栈上（从 SP 指向的位置开始）。

ARM64 (AArch64) 参数传递规则

  1. 前 8 个参数（整数/指针）使用 X0-X7（64 位）传递，32 位值使用 W0-W7。

  2. 超过 8 个参数 存放在栈上（SP 指向的位置）。

2.5 打印寄存器

编写一个方法用于打印所有 ARM64 寄存器的值，包括 通用寄存器（X0-X30）、SP（堆栈指针） 和 PC（程序计数器）。

from unicorn.arm64_const import *
import unicorn


def dump_registers(mu):
    """ 打印 Unicorn ARM64 CPU 的所有寄存器 """
    print("\n====== Registers Dump ======")

    # 遍历 X0 - X30
    for i in range(31):  # X0 ~ X30
        reg_id = getattr(unicorn.arm64_const, f'UC_ARM64_REG_X{i}')
        value = mu.reg_read(reg_id)
        print(f"X{i:02}: 0x{value:016x}")

    # 打印 SP（栈指针）和 PC（程序计数器）
    sp = mu.reg_read(UC_ARM64_REG_SP)
    pc = mu.reg_read(UC_ARM64_REG_PC)

    print(f"\nSP:  0x{sp:016x}")
    print(f"PC:  0x{pc:016x}")
    print("============================\n")

2.6 开始执行

  1. 打印执行前寄存器值

  2. 调用 emu_start ( start_address, end_address ) 开始执行 add 方法的机器码

  3. 读取寄存器 X0 值，打印计算结果

  4. 打印执行后寄存器值

print("\nBefore execution:")
dump_registers(mu)

# **运行 Unicorn**
mu.emu_start(CODE_BASE, CODE_BASE + len(add_function_data))

# **获取结果**
result = mu.reg_read(UC_ARM64_REG_X0)
print(f"add(1, 1) = {result}")

print("\nAfter execution:")
dump_registers(mu)

Unicorn 中读取字符串

Unicorn 中读取指定地址的字符串并支持中文。

from unicorn import Uc, UC_ARCH_ARM64, UC_MODE_ARM
from unicorn.arm64_const import *


def read_string(mu, address, max_length=256):
    """
    Unicorn 中读取以 NULL 结尾的 C 字符串
    :param mu: Unicorn 实例
    :param address: 字符串在内存中的起始地址
    :param max_length: 最大读取长度，防止死循环
    :return: 解析出的字符串
    """
    data = bytearray()

    for _ in range(max_length):
        byte = mu.mem_read(address, 1)[0]  # 读取 1 字节
        if byte == 0:
            break  # 遇到 NULL 终止符，结束读取
        data.append(byte)
        address += 1

    return data.decode("utf-8")  # 解析为字符串


# 初始化 Unicorn ARM64 模拟器
mu = Uc(UC_ARCH_ARM64, UC_MODE_ARM)

# 定义内存布局
MEM_BASE = 0x100000  # 假设基地址
MEM_SIZE = 2 * 1024 * 1024  # 2MB 内存

# 分配内存
mu.mem_map(MEM_BASE, MEM_SIZE)

# 假设字符串存放在 0x101000
STRING_ADDR = MEM_BASE + 0x1000

# 将字符串写入模拟内存
mu.mem_write(STRING_ADDR ,b'Hello, Unicorn!')

# 设置 x0 寄存器为字符串地址
mu.reg_write(UC_ARM64_REG_X0, STRING_ADDR)

# 读取寄存器 x0 并解析字符串
x0_addr = mu.reg_read(UC_ARM64_REG_X0)
parsed_string = read_string(mu, x0_addr)

# 输出字符串
print(f"String at x0: {parsed_string}")

# 将 UTF-8 编码字符串写入模拟内存，并添加 NULL 终止符
mu.mem_write(STRING_ADDR, "Hello, Unicorn中文!".encode('utf-8') + b'\x00')

# 重新解析字符串
parsed_string = read_string(mu, x0_addr)

# 输出字符串
print(f"String at x0: {parsed_string}")
