        logging.info(f"[-] No classes containing '{keyword}' found.")
    else:
        logging.info(f"\n[+] Total {len(found)} matches found.")

    return found


def sort_jar_paths(jar_paths: List[str]) -> List[str]:
    """
    å¯¹åŒ…å« base.apkã€base.apk_classesN.jar çš„è·¯å¾„åˆ—è¡¨è¿›è¡Œæ’åºï¼Œç¡®ä¿ _classes2 æ’åœ¨ _classes10 å‰é¢ã€‚

    :param jar_paths: æœªæ’åºçš„ jar æ–‡ä»¶è·¯å¾„åˆ—è¡¨
    :return: æ’åºåçš„ jar æ–‡ä»¶è·¯å¾„åˆ—è¡¨
    """

    def extract_index(path: str) -> int:
        """
        æå–è·¯å¾„ä¸­ _classesN çš„ N æ•°å­—éƒ¨åˆ†ç”¨äºæ’åºã€‚
        å¦‚æœæ˜¯ base.apk.jar åˆ™è¿”å› 0ï¼Œè¡¨ç¤ºä¼˜å…ˆæ’åºã€‚
        """
        match = re.search(r'_classes(\d+)\.jar$', path)
        if match:
            return int(match.group(1))  # æå– _classesN ä¸­çš„ N
        return 0  # base.apk.jar æ²¡æœ‰ _classesNï¼Œé»˜è®¤æœ€å°å€¼

    # æŒ‰ç…§æå–å‡ºçš„æ•°å­—ç´¢å¼•è¿›è¡Œæ’åº
    return sorted(jar_paths, key=extract_index)


def find_class_and_content_in_jars(directory, keyword):
    """
    åœ¨æŒ‡å®šç›®å½•ä¸‹æ‰€æœ‰ JAR ä¸­æœç´¢ï¼š
    1. ç±»è·¯å¾„ä¸­åŒ…å«å…³é”®å­—çš„ç±»å
    2. ç±»çš„å­—èŠ‚ç ä¸­åŒ…å«å…³é”®å­—å†…å®¹

    :param directory: è¦æœç´¢çš„ç›®å½•
    :param keyword: è¦æŸ¥æ‰¾çš„å…³é”®å­—ï¼ˆæ”¯æŒç±»åè·¯å¾„æˆ–å†…å®¹å…³é”®å­—ï¼‰
    """
    if not keyword:
        logging.info("[-] å…³é”®è¯ä¸èƒ½ä¸ºç©º")
        return

    logging.info(f"[+] Searching for class path or class bytecode containing: {keyword}")

    keyword_bin = keyword.encode()  # è½¬ä¸ºäºŒè¿›åˆ¶ç”¨äºå†…å®¹åŒ¹é…
    keyword_path = keyword.replace('.', '/')

    matched_entries = []
    matched_jars = set()

    for root, _, files in os.walk(directory):
        for file in files:
            if file.endswith(".jar"):
                jar_path = os.path.join(root, file)
                try:
                    with zipfile.ZipFile(jar_path, 'r') as jar:
                        for entry in jar.namelist():
                            if not entry.endswith(".class"):
                                continue

                            matched = False

                            # â‘  ç±»åè·¯å¾„ä¸­åŒ…å«å…³é”®å­—
                            if keyword_path in entry:
                                logging.info(f"[âœ“] Keyword in class name: {entry} ({jar_path})")
                                matched = True

                            # â‘¡ å­—èŠ‚ç ä¸­åŒ…å«å…³é”®å­—ï¼ˆå¦‚å­—ç¬¦ä¸²å¸¸é‡ï¼‰
                            try:
                                with jar.open(entry) as class_file:
                                    content = class_file.read()
                                    if keyword_bin in content:
                                        logging.info(f"[âœ“] Keyword in class bytecode: {entry} ({jar_path})")
                                        matched = True
                            except Exception as e:
                                logging.info(f"[!] Failed reading {entry} in {jar_path}: {e}")

                            if matched:
                                matched_entries.append((jar_path, entry))
                                matched_jars.add(jar_path)

                except zipfile.BadZipFile:
                    logging.info(f"[!] Skipping corrupted jar: {jar_path}")

    if not matched_entries:
        logging.info(f"[-] No match found for keyword '{keyword}'")
    else:
        logging.info(f"\n[+] Total {len(matched_entries)} match(es) found.")
        logging.info(f"[+] Matched JAR count: {len(matched_jars)}")
        logging.info("[+] Matched JAR files:")
        for jar_file in sort_jar_paths(matched_jars):
            logging.info(f"    - {jar_file}")
